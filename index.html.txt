<script>
    function save() {
        localStorage.setItem('eb_n1', document.getElementById('name1').value);
        localStorage.setItem('eb_n2', document.getElementById('name2').value);
        localStorage.setItem('eb_b1', document.getElementById('bday1').value);
        localStorage.setItem('eb_b2', document.getElementById('bday2').value);
        localStorage.setItem('eb_bio', document.getElementById('bio').value);
    }

    function load() {
        document.getElementById('name1').value = localStorage.getItem('eb_n1') || '';
        document.getElementById('name2').value = localStorage.getItem('eb_n2') || '';
        document.getElementById('bday1').value = localStorage.getItem('eb_b1') || '';
        document.getElementById('bday2').value = localStorage.getItem('eb_b2') || '';
        document.getElementById('bio').value = localStorage.getItem('eb_bio') || '';
        render();
    }

    let items = JSON.parse(localStorage.getItem('eb_items')) || [];
    function addItem() {
        const n = document.getElementById('mName').value;
        const p = document.getElementById('mPrice').value;
        if(n) {
            items.unshift({n, p, d: new Date().toLocaleDateString()});
            localStorage.setItem('eb_items', JSON.stringify(items));
            render();
            document.getElementById('mName').value = '';
            document.getElementById('mPrice').value = '';
        }
    }

    function render() {
        document.getElementById('list').innerHTML = items.map(i => 
            `<div class="record-item"><span><b>${i.n}</b><br><small style="color:#777">${i.d}</small></span><span style="color:var(--emi-orange)">${i.p?'$'+i.p:''}</span></div>`
        ).join('');
    }

    let T;
    let s = parseInt(localStorage.getItem('eb_s')) || 0;
    function upT() {
        let h=Math.floor(s/3600), m=Math.floor((s%3600)/60), sec=s%60;
        document.getElementById('timer').innerText = `${h.toString().padStart(2,'0')}:${m.toString().padStart(2,'0')}:${sec.toString().padStart(2,'0')}`;
    }
    function runTimer() {
        const b = document.getElementById('tBtn');
        if(b.innerText === "é–‹å•Ÿè´è¶æ™‚å…‰") {
            b.innerText = "æš«åœæ™‚å…‰"; b.style.borderColor = "var(--emi-orange)";
            T = setInterval(()=>{ s++; upT(); localStorage.setItem('eb_s', s); }, 1000);
        } else {
            b.innerText = "é–‹å•Ÿè´è¶æ™‚å…‰"; b.style.borderColor = "var(--bonnie-purple)";
            clearInterval(T);
        }
    }
